namespace methodtask2
{
    internal class Program
    {
        static void Main(string[] args)
        {
            string cumle = Console.ReadLine();
            IlkHerf(cumle);
        }
        static void IlkHerf(string cumle)
        {
            bool baslangic = true; 
            for (int i = 0; i < cumle.Length; i++)
            {
                if ( baslangic&&cumle[i] != ' ')
                {
                    Console.Write(cumle[i]+" "); 
                    baslangic = false; 
                }
                else if (cumle[i] == ' ')
                {
                    baslangic = true; 
                }
            }
            Console.WriteLine();
        }
    }
